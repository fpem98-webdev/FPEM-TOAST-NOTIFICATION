Référence API

Documentation complète de toutes les méthodes et options disponibles

notify(type, options)
Méthode principale pour créer et afficher une notification
fpemNotif.notify('success', { title: 'Titre', message: 'Message' })

Paramètres

Paramètre Type Description Par défaut
type string Type de notification: 'success', 'error', 'warning', 'info', 'custom' 'custom'
options.title string Titre de la notification ''
options.message string Message principal ''
options.icon string Icône à afficher (emoji, caractère) Auto selon type
options.duration number Durée d'affichage en millisecondes 4000
options.persistent boolean Notification persistante (pas de fermeture auto) false
options.showCloseButton boolean Afficher le bouton de fermeture false
options.actions array Actions personnalisées [{label, callback, primary}] []
create(type)
Crée un builder pour construire une notification de manière fluide
fpemNotif.create('info').title('Titre').message('Message').show()

Méthodes du Builder

Méthode Paramètre Description
.title(text) string Définit le titre
.message(text) string Définit le message
.icon(icon) string Définit l'icône
.duration(ms) number Définit la durée
.persistent(bool) boolean Rend persistante
.closable(bool) boolean Affiche bouton fermeture
.action(label, callback) string, function Ajoute une action
.show() - Affiche la notification
Méthodes Utilitaires
Méthodes avancées pour cas d'usage spécifiques
Méthode Description Retour
loading(options) Notification de chargement avec contrôleur Objet de contrôle
confirm(options) Boîte de confirmation avec Promise Promise<boolean>

update(id, options) Met à jour une notification existante HTMLElement

// Validation de formulaire avec notifications
function validateForm(formData) {
  const errors = [];
  
  if (!formData.email) {
    errors.push('Email requis');
  } else if (!isValidEmail(formData.email)) {
    errors.push('Email invalide');
  }
  
  if (!formData.password || formData.password.length < 8) {
    errors.push('Mot de passe trop court (min 8 caractères)');
  }
  
  if (errors.length > 0) {
    fpemNotif.error({
      title: 'Erreurs de validation',
      message: errors.join('\n'),
      persistent: true,
      showCloseButton: true,
      actions: [
        {
          label: 'Corriger',
          callback: () => focusFirstError()
        }
      ]
    });
    return false;
  }
  
  // Succès
  fpemNotif.success({
    title: 'Formulaire valide',
    message: 'Toutes les informations sont correctes',
    icon: '✅'
  });
  return true;
}

// Sauvegarde avec feedback
async function saveForm(data) {
  const loader = fpemNotif.loading({
    title: 'Sauvegarde',
    message: 'Enregistrement en cours...'
  });
  
  try {
    await submitFormData(data);
    loader.success('Données sauvegardées !');
  } catch (error) {
    loader.error('Erreur lors de la sauvegarde');
    
    // Notification d'erreur détaillée
    fpemNotif.error({
      title: 'Erreur de sauvegarde',
      message: error.message,
      persistent: true,
      actions: [
        { label: 'Réessayer', callback: () => saveForm(data) },
        { label: 'Signaler', callback: () => reportError(error) }
      ]
    });
  }
}

// Upload de fichier avec progression détaillée
async function uploadFile(file) {
  // Validation du fichier
  if (file.size > 10 * 1024 * 1024) {
    fpemNotif.warning({
      title: 'Fichier trop volumineux',
      message: 'Taille maximale autorisée: 10 MB',
      icon: '⚠️'
    });
    return;
  }
  
  // Notification de démarrage
  const loader = fpemNotif.loading({
    title: `Upload: ${file.name}`,
    message: 'Préparation du téléchargement...',
    id: `upload-${file.name}`
  });
  
  const formData = new FormData();
  formData.append('file', file);
  
  try {
    const response = await fetch('/upload', {
      method: 'POST',
      body: formData,
      onUploadProgress: (progressEvent) => {
        const progress = Math.round(
          (progressEvent.loaded * 100) / progressEvent.total
        );
        
        loader.updateProgress(progress, 
          `Upload: ${progress}% (${formatBytes(progressEvent.loaded)}/${formatBytes(progressEvent.total)})`
        );
      }
    });
    
    if (response.ok) {
      const result = await response.json();
      loader.success(`${file.name} uploadé avec succès !`);
      
      // Notification avec actions post-upload
      setTimeout(() => {
        fpemNotif.info({
          title: 'Fichier prêt',
          message: 'Votre fichier est maintenant disponible',
          actions: [
            { label: 'Voir', callback: () => window.open(result.url) },
            { label: 'Partager', callback: () => shareFile(result.url) }
          ]
        });
      }, 2000);
      
    } else {
      throw new Error('Erreur serveur');
    }
    
  } catch (error) {
    loader.error(`Échec de l'upload: ${error.message}`);
  }
}

// Upload multiple avec gestion de la file
function uploadMultipleFiles(files) {
  const totalFiles = files.length;
  let completed = 0;
  
  fpemNotif.info({
    title: 'Upload groupé',
    message: `${totalFiles} fichier(s) en attente`,
    id: 'batch-upload'
  });
  
  files.forEach(async (file, index) => {
    await uploadFile(file);
    completed++;
    
    fpemNotif.update('batch-upload', {
      message: `Progression: ${completed}/${totalFiles} fichiers`
    });
    
    if (completed === totalFiles) {
      setTimeout(() => {
        fpemNotif.update('batch-upload', {
          type: 'success',
          title: 'Upload terminé',
          message: `Tous les fichiers ont été uploadés !`,
          persistent: false,
          duration: 3000
        });
      }, 1000);
    }
  });
}
JavaScript - Notifications temps réel 
// WebSocket pour notifications temps réel
class RealtimeNotifications {
  constructor() {
    this.socket = new WebSocket('wss://api.example.com/notifications');
    this.setupSocketListeners();
  }
  
  setupSocketListeners() {
    this.socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleNotification(data);
    };
    
    this.socket.onopen = () => {
      fpemNotif.success({
        title: 'Connecté',
        message: 'Notifications temps réel activées',
        icon: '🔗',
        duration: 2000
      });
    };
    
    this.socket.onclose = () => {
      fpemNotif.warning({
        title: 'Connexion fermée',
        message: 'Tentative de reconnexion...',
        persistent: true,
        id: 'connection-lost'
      });
      
      // Tentative de reconnexion
      setTimeout(() => this.reconnect(), 3000);
    };
  }
  
  handleNotification(data) {
    switch (data.type) {
      case 'message':
        fpemNotif.info({
          title: `Nouveau message de ${data.from}`,
          message: data.content,
          icon: '💬',
          actions: [
            { label: 'Répondre', callback: () => openChat(data.from) },
            { label: 'Marquer lu', callback: () => markAsRead(data.id) }
          ]
        });
        break;
        
      case 'system_alert':
        fpemNotif.error({
          title: 'Alerte système',
          message: data.message,
          persistent: true,
          showCloseButton: true,
          actions: [
            { label: 'Détails', callback: () => showAlertDetails(data.id) }
          ]
        });
        break;
        
      case 'friend_online':
        fpemNotif.create('custom')
          .title('Ami connecté')
          .message(`${data.username} vient de se connecter`)
          .icon('🟢')
          .duration(3000)
          .action('Dire bonjour', () => sendGreeting(data.user_id))
          .show();
        break;
        
      case 'achievement':
        fpemNotif.success({
          title: '🏆 Succès débloqué !',
          message: data.achievement_name,
          icon: '🎉',
          duration: 6000,
          actions: [
            { label: 'Voir', callback: () => showAchievement(data.id) },
            { label: 'Partager', callback: () => shareAchievement(data.id) }
          ]
        });
        break;
    }
  }
  
  reconnect() {
    this.socket = new WebSocket('wss://api.example.com/notifications');
    this.setupSocketListeners();
  }
}

// Système de notifications push du navigateur
function requestNotificationPermission() {
  if ('Notification' in window) {
    Notification.requestPermission().then(permission => {
      if (permission === 'granted') {
        fpemNotif.success({
          title: 'Notifications autorisées',
          message: 'Vous recevrez des notifications même quand l\'onglet est fermé'
        });
      } else {
        fpemNotif.info({
          title: 'Notifications limitées',
          message: 'Seules les notifications dans l\'application seront affichées'
        });
      }
    });
  }
}

// Intégration notifications navigateur + FPEM
function showBrowserNotification(title, message, actions = []) {
  // Notification navigateur si permission accordée
  if (Notification.permission === 'granted') {
    const notification = new Notification(title, {
      body: message,
      icon: '/icon.png'
    });
    
    notification.onclick = () => {
      window.focus();
      notification.close();
    };
  }
  
  // Notification FPEM si onglet actif
  if (!document.hidden) {
    fpemNotif.info({
      title,
      message,
      actions: actions.map(action => ({
        label: action.label,
        callback: action.callback
      }))
    });
  }
}
